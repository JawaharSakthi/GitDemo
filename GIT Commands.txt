git config --global -l --> To check the username and email
git init --> To create local repository or staging area
git status --> To check the status of the files
git add --> add the files to staging area
git diff --> changes made will be displayed
git restore <filename>  --> restore the deleted files
git commit -m "message" --> store in the local repository
git log --> shows the history, checks the user who added previously
git remote add origin "paste the url" --> Establish the connection between local and global repository
